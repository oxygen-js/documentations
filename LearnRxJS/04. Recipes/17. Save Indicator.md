# Save Indicator

Этот рецепт демонстрирует создание индикатора сохранения в духе Google Docs с
помощью RxJS.



### Пример кода

( [StackBlitz](https://stackblitz.com/edit/rxjs-3txbvy?file=index.ts) )

![Save Indicator](https://drive.google.com/uc?export=view&id=1sYFqLoKlT0EPHxSDMSX7pT14RDjybU0Q)

```js
import { fromEvent, of, merge, empty, concat, defer } from 'rxjs';
import {
  delay,
  map,
  mergeMap,
  tap,
  debounceTime,
  distinctUntilChanged,
  mapTo,
  filter,
  share,
  switchAll
} from 'rxjs/operators';
import { format } from 'date-fns';

// отслеживаем текущие сохранения
let savesInProgress = 0;

// ссылки
const input = document.getElementById('note-input');
const saveIndicator = document.querySelector('.save-indicator');

// потоки
const keyup$ = fromEvent(input, 'keyup');

// фейковый запрос на сохранение
const saveChanges = value => {
  return of(value).pipe(delay(1500));
};

/**
 * Запускаем сохранение, когда пользователь перестаёт печатать на 200мс.
 * После того как новые данные успешно сохранены, показываем индикатор
 * сохранения и время последнего обновления.
 */
const inputToSave$ = keyup$.pipe(
  debounceTime(200),
  map(e => e.target.value),
  distinctUntilChanged(),
  share()
);

const savesInProgress$ = inputToSave$.pipe(
  mapTo(of('Saving')),
  tap(_ => savesInProgress++)
);

const savesCompleted$ = inputToSave$.pipe(
  mergeMap(saveChanges),
  tap(_ => savesInProgress--),
  // игнорируем, если выполняются дополнительные сохранения
  filter(_ => !savesInProgress),
  mapTo(
    concat(
      // показываем «Saved» в течение 2с
      of('Saved!'),
      empty().pipe(delay(2000)),
      // затем время последнего обновления, откладываем через defer для корректного времени
      defer(() => of(`Last updated: ${format(Date.now(), 'MM/DD/YYYY hh:mm')}`))
    )
  )
);

merge(savesInProgress$, savesCompleted$)
  .pipe(
    /*
   Если новое сохранение приходит, когда работает наш observable завершения, мы хотим переключиться на него для обновления статуса.
  */
    switchAll()
  )
  .subscribe(status => {
    saveIndicator.innerHTML = status;
  });
```
