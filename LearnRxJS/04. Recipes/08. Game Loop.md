# Игровой цикл

_Автор: [@barryrowe](https://twitter.com/barryrowe)_

Этот рецепт демонстрирует один из способов, которым вы можете создать игровой цикл как
объединённый набор потоков. Рецепт призван показать, как можно переосмыслить существующие
задачи с помощью реактивного подхода. В этом рецепте мы представляем весь цикл в виде
потока кадров и их deltaTime с момента предыдущего кадра. С этим объединяется
поток пользовательского ввода и текущее состояние игры (gameState), которое мы можем использовать, чтобы
обновлять наши объекты и отрисовывать их на экране при испускании каждого кадра.

```js
import { BehaviorSubject, Observable, of, fromEvent } from 'rxjs';
import { buffer, bufferCount, expand, filter, map,  share, tap, withLatestFrom } from 'rxjs/operators';

import { IFrameData } from './frame.interface';
import { KeyUtil } from './keys.util';
import { clampMag, runBoundaryCheck, clampTo30FPS } from './game.util';

const boundaries = {
  left: 0,
  top: 0,
  bottom: 300,
  right: 400
};
const bounceRateChanges = {
  left: 1.1,
  top: 1.2,
  bottom: 1.3,
  right: 1.4
}
const baseObjectVelocity = {
  x: 30,
  y: 40,
  maxX: 250,
  maxY: 200
};

const gameArea: HTMLElement = document.getElementById('game');
const fps: HTMLElement = document.getElementById('fps');

/**
 * Это основная логика нашего игрового цикла. Здесь мы обновляем наши объекты и gameState
 * на каждом кадре. Переданный deltaTime указан в секундах, нам дают текущее состояние
 * и любые состояния ввода (inputStates). Возвращает обновлённое состояние игры.
 */
const update = (deltaTime: number, state: any, inputState: any): any => {
  //console.log("Input State: ", inputState);
  if(state['objects'] === undefined) {
    state['objects'] = [
      {
        // Свойства трансформации
        x: 10, y: 10, width: 20, height: 30,
        // Свойства состояния
        isPaused: false, toggleColor: '#FF0000', color: '#000000',
        // Свойства движения
        velocity: baseObjectVelocity
      },
      {
        // Свойства трансформации
        x: 200, y: 249, width: 50, height: 20,
        // Свойства состояния
        isPaused: false, toggleColor: '#00FF00', color: '#0000FF',
        // Свойства движения
        velocity: {x: -baseObjectVelocity.x, y: 2*baseObjectVelocity.y} }
    ];
  } else {

    state['objects'].forEach((obj) => {
      // Обрабатываем ввод
      if (inputState['spacebar']) {
        obj.isPaused = !obj.isPaused;
        let newColor = obj.toggleColor;
        obj.toggleColor = obj.color;
        obj.color = newColor;
      }

      // Обрабатываем обновления игрового цикла
      if(!obj.isPaused) {

        // Применяем перемещения по скорости
        obj.x = obj.x += obj.velocity.x*deltaTime;
        obj.y = obj.y += obj.velocity.y*deltaTime;

        // Проверяем, не вышли ли мы за границы
        const didHit = runBoundaryCheck(obj, boundaries);
        // Обрабатываем коррекции по границам
        if(didHit){
          if(didHit === 'right' || didHit === 'left') {
            obj.velocity.x *= -bounceRateChanges[didHit];
          } else {
            obj.velocity.y *= -bounceRateChanges[didHit];
          }
        }
      }

      // Ограничиваем скорости на случай, если отскоки от границ разогнали
      //  нас слишком быстро.
      obj.velocity.x = clampMag(obj.velocity.x, 0, baseObjectVelocity.maxX);
      obj.velocity.y = clampMag(obj.velocity.y, 0, baseObjectVelocity.maxY);
    });
  }

  return state;
}

/**
 * Это наша функция отрисовки. Мы берём заданное состояние игры и отрисовываем элементы
 * на основе их последних свойств.
 */
const render = (state: any) => {
  const ctx: CanvasRenderingContext2D = (<HTMLCanvasElement>gameArea).getContext('2d');
  // Очищаем холст
  ctx.clearRect(0, 0, gameArea.clientWidth, gameArea.clientHeight);

  // Отрисовываем все наши объекты (для простоты — обычные прямоугольники)
  state['objects'].forEach((obj) => {
    ctx.fillStyle = obj.color;
    ctx.fillRect(obj.x, obj.y, obj.width, obj.height);
  });
};



/**
 * Эта функция возвращает observable, который испустит следующий кадр, как только
 * браузер вернёт шаг анимационного кадра. По предыдущему кадру она вычисляет
 * дельту времени, и мы также ограничиваем её до 30 FPS на случай, если кадры окажутся длинными.
 */
const calculateStep: (prevFrame: IFrameData) => Observable<IFrameData> = (prevFrame: IFrameData) => {
  return Observable.create((observer) => {

    requestAnimationFrame((frameStartTime) => {
      // Миллисекунды в секунды
      const deltaTime = prevFrame ? (frameStartTime - prevFrame.frameStartTime)/1000 : 0;
      observer.next({
        frameStartTime,
        deltaTime
      });
    })
  })
  .pipe(
    map(clampTo30FPS)
  )
};

// Это наш основной поток кадров. Мы используем expand, чтобы рекурсивно вызывать функцию
//  `calculateStep` выше, которая будет давать нам каждый новый кадр на основе
//  вызовов window.requestAnimationFrame. Expand испускает значение observable, возвращаемого
//  вызванной функцией, а также рекурсивно вызывает функцию с этим же испущенным значением.
//  Это идеально подходит для вычисления шагов кадров, потому что каждый шаг
//  должен знать lastStepFrameTime, чтобы вычислить следующий. Мы также хотим запрашивать
//  новый кадр только после того, как вернётся текущий запрошенный кадр.
const frames$ = of(undefined)
  .pipe(
    expand((val) => calculateStep(val)),
    // Expand испускает первое переданное ему значение, и в этом
    //  случае мы просто хотим проигнорировать undefined-кадр на входе
    filter(frame => frame !== undefined),
    map((frame: IFrameData) => frame.deltaTime),
    share()
  )

// Это наш основной поток событий keyDown. Он испускает объект вроде `{"spacebar": 32}`
//  каждый раз, когда нажата клавиша.
const keysDown$ = fromEvent(document, 'keydown')
  .pipe(
    map((event: KeyboardEvent) => {
      const name = KeyUtil.codeToKey(''+event.keyCode);
      if (name !== ''){
        let keyMap = {};
        keyMap[name] = event.code;
        return keyMap;
      } else {
        return undefined;
      }
    }),
    filter((keyMap) => keyMap !== undefined)
  );

// Здесь мы буферизуем наш поток keyDown, пока не получим испускание нового кадра. Это
//  даёт нам набор всех событий keyDown, сработавших с момента предыдущего
//  кадра. Мы сводим их все к единому словарю нажатых клавиш.
const keysDownPerFrame$ = keysDown$
  .pipe(
    buffer(frames$),
    map((frames: Array<any>) => {
      return frames.reduce((acc, curr) => {
        return Object.assign(acc, curr);
      }, {});
    })
  );

// Поскольку мы будем обновлять состояние игры на каждом кадре, мы можем использовать Observable,
//  чтобы отслеживать его как последовательность состояний, где последнее испускание — это текущее
//  состояние нашей игры.
const gameState$ = new BehaviorSubject({});

// Здесь мы запускаем нашу игру!
//  Мы подписываемся на поток frames$, чтобы запустить его, и обязательно
//  подмешиваем последнее испускание из потока ввода, чтобы получить данные,
//  необходимые для выполнения обновлений gameState.
frames$
  .pipe(
    withLatestFrom(keysDownPerFrame$, gameState$),
    // ДОМАШНЕЕ ЗАДАНИЕ: обработайте key-up и отобразите его в настоящий объект изменения состояния клавиши
    map(([deltaTime, keysDown, gameState]) => update(deltaTime, gameState, keysDown)),
    tap((gameState) => gameState$.next(gameState))

  )
  .subscribe((gameState) => {
    render(gameState);
  });


// Усредняем каждые 10 кадров, чтобы вычислить наш FPS
frames$
  .pipe(
    bufferCount(10),
    map((frames) => {
      const total = frames
        .reduce((acc, curr) => {
          acc += curr;
          return acc;
        }, 0);

        return 1/(total/frames.length);
    })
  ).subscribe((avg) => {
    fps.innerHTML = Math.round(avg) + '';
  })
```
